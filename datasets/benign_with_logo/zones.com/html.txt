	function js_scrollTo(dest) {
		var hash = window.location.hash;
		var $dest = dest ? $(dest) : $(hash);

		if ($dest.length === 0) return false;

		var options = {
			dist: $dest.attr('data-dist') ? $dest.attr('data-dist') : 0,
			speed: $dest.attr('data-speed') ? $dest.attr('data-speed') : 500
		}

		setTimeout(function () {
			$('html, body').animate({
				'scrollTop': $dest.offset().top - options['dist']
			}, options['speed']);
		}, 500);

		return false;
	}

function HeaderContext(el) {
    this.$el = $(el);
    this.historyStore  = "zones_customer_search_history";
    this.smallScreenSize = 768;
    this.mediumScreenSize= 992;
    this.isTouchDevice   = $("html.touch").length > 0 ? true:false;
}

HeaderContext.prototype = {
    init: function() {
        var self = this, $body = $("body");
        var $form = self.$el.find("form[name='search']");

        self.$el.on('focus',"input[name='txt_search']", function() {
            $body.addClass("search-active");
        });

        self.$el.on("click","#mobile-hamburger-menu", function() {
            $body.removeClass("search-active").addClass("mobile-menu--open");
            return false;
        })

        self.$el.find("#mobile-menu--close-btn").on("click", function() {
            $body.removeClass("mobile-menu--open");
            return false;
        })

        self.$el.find(".mobile-menu--dropdownIcon, .mobile-menu--dropdownIcon a").on("click touch", function() {
            if (window.innerWidth <= self.mediumScreenSize || self.isTouchDevice) {
                console.log("small")
                var $this   = $(this);
                var target  = $this.attr("data-target");
                if (target === null) return false; 
                
                var $target = $(target);
                console.log($target, $target.length)
                $target.toggleClass("active");
                return false;
            } 
        });
        
        //handle click outsite searchbox
        $(document).on('click', function(e) {
            var $this = $(this);
            var target= $(e.target);
            if (
                $body.hasClass("search-active") &&
                !target.is("input[name='txt_search']") &&
                !target.is(".clear-searchterm")
            ) {
                $(".search-active").addClass("removing")
                setTimeout(function() {
                    $body.removeClass("search-active removing");
                },500);
            }   
        });
    },
    
    getLocalStorage: function() {
        var store = localStorage.getItem(this.historyStore);
        if (this.isJSON(store)) {
            return JSON.parse(store);
        } else {
            return [];
        }
    },

    isJSON: function(str) {
        try {
            JSON.parse(str);
        } catch(e) {
            return false;
        }
        return true;
    }
}

$(function() {
    window.headerContext = new HeaderContext("header.main-header");
    headerContext.init();
})
