    searchForm.addEventListener('submit', function(e) {

            document.body.classList.contains('enable-dates-modal')
            && !bypassCheck
            && (!dateFromEl.value || !dateToEl.value)

            e.preventDefault();

                modalEl = jQuery('#dates_modal');
                modalEl.on('hide.bs.modal', function(e) {
