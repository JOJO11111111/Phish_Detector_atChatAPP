        // wColor = wedge background color
        // oColor = wedge outline color
        animateWedge(wedge, val, showVal) {

            // val is a percentage
            let shadedDeg = val * 90 / 100;

            // first draw a transparent background arc
            // - we want to be able to click on unshaded area as well as shaded
            let bg = this.drawArcFromTo(wedge.x, wedge.y, this.radius / 2, wedge.start, wedge.end, wedge.bgColor, this.radius);
            let l1 = this.drawLineAt(wedge.x, wedge.y, this.radius, wedge.start, wedge.oColor, this.strokeWidth);

                    let path = this.drawArcFromTo(wedge.x, wedge.y, this.radius / 2, wedge.start, wedge.start + val, wedge.wColor, this.radius);
                    this.animatingPaths.push(path);

                    // keep our center circle at the front
                    this.drawCenterCircle(this.centerX, this.centerY, this.mastery);

                    this.animatingPaths.forEach(function (path) {
                        path.remove();
                    });
                    this.animatingPaths = [];

                    this.drawWedge(wedge, shadedDeg, val, showVal, bg, l2, a1, l1);

            }

        drawWedge(wedge, shadedDeg, val, showVal, bg, l2, a1, l1) {
            // redraw the final wedge shading
            let w = this.drawArcFromTo(wedge.x, wedge.y, this.radius / 2, wedge.start, wedge.start + shadedDeg, wedge.wColor, this.radius);

            // we won't add the text to the group as adding a shadow is distracting
            if (this.showText) {
                this.drawWedgeText(wedge.tx, wedge.ty, wedge.title, wedge.multilineTitle, val, showVal, wedge.tColor);
            }

            // add everything to a group and assign the id
            let g = this.snap.group(bg, w, l2, a1, l1);
            g.attr({id: "w_" + wedge.id});

            g.click(this.wedgeClick.bind(this));

                let arc = this.snap.select('#a1_' + gId);
                arc.animate({
                    stroke: arc.attr("color")
                }, 200);
            }.bind(this));


            // keep our center circle at the front
            this.drawCenterCircle(this.centerX, this.centerY, this.mastery);

            let statusEvent = new CustomEvent('wedge_ready', {'detail': {id: wedge.id}});
            document.body.dispatchEvent(statusEvent);
        }

        drawWedgeText(x, y, title, multiline, val, showVal, color) {
            // if it's multiline, we do it differently
            if (!multiline) {
                title += "  ";
            }

            // val is the percentage from 0 - 100
            // if it's -1, we won't show it
            let valText = showVal ? val + "%" : "";

            let t = this.snap.paper.text(x, y, [title, valText])
                .attr({'textAnchor': 'middle', 'alignmentBaseline': 'central'});

            t.selectAll("tspan").forEach(function (tspan, i) {
                switch (i) {
                    case 0:
                        tspan.attr({fill: color, fontWeight: "bold", fontSize: "20px"});
                        if (multiline) {
                            tspan.attr({x: x, y: y - 10});
                        }
                        break;
                    case 1:
                        tspan.attr({fill: this.pcColor, fontSize: "15px"});
                        if (multiline) {
                            tspan.attr({x: x, y: y + 10});
                        }
                        break;
                }
            }.bind(this));

            return t;
        }

        drawArcFromTo(centerX, centerY, radius, startDeg, endDeg, color, strokeWidth) {
            let largeArc = (endDeg - startDeg) > 180 ? 1 : 0;
            if (startDeg < 90) startDeg += 360;
            if (endDeg < 90) endDeg += 360;
            let startRad = Math.PI * (startDeg - 90) / 180;
            let endRad = Math.PI * (endDeg - 90) / 180;
            let startx = centerX + radius * Math.cos(startRad);
            let starty = centerY + radius * Math.sin(startRad);
            let endx = centerX + radius * Math.cos(endRad);
            let endy = centerY + radius * Math.sin(endRad);

            let path = "M" + startx + "," + starty + " A" + radius + "," + radius + " 0 " + largeArc + ",1 " + endx + "," + endy;

            let arc = this.snap.path(path);
            arc.attr({
                stroke: color,
                fill: 'none',
                strokeWidth: strokeWidth
            });

            return arc;
        }

        drawLineAt(startx, starty, radius, deg, color, strokeWidth) {

            let endRad = Math.PI * (deg - 90) / 180;
            let endx = startx + radius * Math.cos(endRad);
            let endy = starty + radius * Math.sin(endRad);

            let line = this.snap.line(startx, starty, endx, endy);
            line.attr({
                stroke: color,
                strokeWidth: strokeWidth,
                strokeLinecap: "square"
            });

            return line;
        }

        drawCenterCircle(x, y, val) {
            // zero value tells us to hide it
            if (val === 0) return;

            // if we already have them bring them to the front
            if (this.centerCircle && this.centerText) {
                this.centerCircle.node.parentElement.appendChild(this.centerCircle.node);
                this.centerText.node.parentElement.appendChild(this.centerText.node);
                return;
            }

            // otherwise create them

            this.centerCircle = this.snap.paper.circle(x, y, 13).attr({fill: "#b3567f", opacity: 1});

            this.centerText = this.snap.paper.text(x, y + 4, val).attr({
                // id: "t_"+id,
                // 'display': 'none',
                fill: 'white',
                'font-size': 14,
                "font-weight": "bold",
                'textAnchor': 'middle',
                // 'alignmentBaseline': 'central' --> doesn't work on Firefox; use y instead
            });
        }

        // reset each wedge to default
        resetWedges() {
            for (let id of ["k", "g", "w", "sp"]) {
                // alert(x);
                let g = this.snap.select('#w_' + id);
                if (g.attr("selected") === "yes") {
                    this.resetWedge(id);
                }
            }
        }

        // when a wedge is clicked, we'll reset all the others, and toggle the one that's clicked
        wedgeClick(e) {
            // we want the parent group - that's where the id is
            // parent id is of the form w_x
            // we want to use x, to make it easy to highlight with an external command
            let id = e.target.parentElement.id.substr(2);
            let g = this.snap.select('#w_' + id);
            let currentState = g.attr("selected");

            this.resetWedges();

            (currentState === "yes") ? this.resetWedge(id) : this.highlightWedge(id);

            let statusEvent = new CustomEvent('wedge_clicked', {'detail': {area: id, selecting: (currentState !== "yes")}});
            document.body.dispatchEvent(statusEvent);
        }

        highlightWedge(id) {
            let g = this.snap.select('#w_' + id);
            if (g.attr("selected") === "yes") return;

            let newMatrix = new Snap.Matrix();
            newMatrix.scale(1.1, 1.1, this.centerX, this.centerY);
            g.animate({transform: newMatrix}, 500, mina.bounce);
            g.attr({filter: this.shadow, selected: "yes"});
        }

        resetWedge(id) {
            let g = this.snap.select('#w_' + id);
            if (g.attr("selected") === "no") return;

            let newMatrix = new Snap.Matrix();
            newMatrix.scale(1, 1, this.centerX, this.centerY);
            g.animate({transform: newMatrix}, 500, mina.bounce);
            g.attr({filter: null, selected: "no"});
        }

        toggleWedge(id) {
            let g = this.snap.select('#w_' + id);
            let currentState = g.attr("selected");
            (currentState === "yes") ? this.resetWedge(id) : this.highlightWedge(id);
        }
    }
