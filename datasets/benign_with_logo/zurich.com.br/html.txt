<script>
function socialShareIteration(ulTag, liTag) {
    var httpContextURL = encodeURIComponent(window.location.href).toLowerCase();
    var listItems = $(ulTag.toString() + ' ' + liTag.toString());

    listItems.each(function (index, value) {
        var hrefValue = $(this).find('a').attr('href');
        if (hrefValue != undefined) {
            if (hrefValue.toLowerCase().indexOf('facebook') > 0 || hrefValue.toLowerCase().indexOf('google') > 0 || hrefValue.toLowerCase().indexOf('xing') > 0) {
                formatURL($(this), hrefValue, httpContextURL);
            }
            else if (hrefValue.toLowerCase().indexOf('linkedin') > 0) {
                var socialURL = hrefValue.split("url=")[1].split("&title")[0];
                var title = hrefValue.split("&title=")[1].split("&summary")[0];
                var summary = hrefValue.split("&summary=")[1].split("&source")[0];
                var source = hrefValue.split("&source=")[1];
                var currentPageSummary = $('.mod-article-header .copy-large:last').text().trim().split(' ').join('+');
                var currentPageTitle = $('.mod-article-header h1').text().trim().split(' ').join('+');

                if (httpContextURL != socialURL) {
                    if (summary.toLowerCase() != currentPageSummary.toLowerCase()) {
                        summary = currentPageSummary;
                    }
                    if (title.toLowerCase() != currentPageTitle.toLowerCase()) {
                        title = currentPageTitle;
                    }
                    var formattedURL = hrefValue.split("url=")[0] + "url=" + httpContextURL + "&title=" + title + "&summary=" + summary + "&source=" + source;
                    $(this).find('a').attr('href', formattedURL);
                }
            }
            else if (hrefValue.toLowerCase().indexOf('pinterest') > 0) {
                var socialURL = hrefValue.split("url=")[1].split("&media")[0];
                var media = hrefValue.split("&media=")[1].split("&description")[0].toLowerCase();
                var description = hrefValue.split("&description=")[1];
                var currentPageMedia = encodeURIComponent(window.location.origin + $('.media--bottom .mod-image__media img').attr('src').split('?')[0]).toLowerCase();
                var currentPageDescription = $('.mod-article-header .copy-large:last').text().trim().split(' ').join('+');

                if (httpContextURL != socialURL) {
                    if (description.toLowerCase() != currentPageDescription.toLowerCase()) {
                        description = currentPageDescription;
                    }
                    if (media != currentPageMedia) {
                        media = currentPageMedia;
                    }
                    var formattedURL = hrefValue.split("url=")[0] + "url=" + httpContextURL + "&media=" + media + "&description=" + description;
                    $(this).find('a').attr('href', formattedURL);
                }
            }
            else if (hrefValue.toLowerCase().indexOf('twitter.com') > 0) {
                var text = hrefValue.split("text=")[1].split("&url")[0].toLowerCase();
                var socialURL = hrefValue.split("&url=")[1];
                var currentPageText = $('.mod-article-header .copy-large:last').text().trim().split(' ').join('+');

                if (httpContextURL != socialURL) {
                    if (text.toLowerCase() != currentPageText.toLowerCase()) {
                        text = currentPageText;
                    }
                    var formattedURL = hrefValue.split("text=")[0] + "text=" + text + "&url=" + httpContextURL;
                    $(this).find('a').attr('href', formattedURL);
                }
            }
        }
    });
}


function formatURL($this, hrefValue, httpContextURL) {
    var socialURL = hrefValue.split("url=")[1];
    if (httpContextURL != socialURL) {
        var formattedURL = hrefValue.substring(0, hrefValue.indexOf("=") + 1) + httpContextURL;
        $this.find('a').attr('href', formattedURL);
    }
}

window.onload = function () {
    if ($('.mod-share-bar__inner li').length > 0) {
        var socialShare = socialShareIteration('.mod-share-bar__inner', 'li');
        console.log("Update Share bar links: Done");
    }
}
