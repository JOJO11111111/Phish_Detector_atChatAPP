var newsletterSignup = function() {
  "use strict";

  var $container = $('.newsletter-signup__form'),
      $form = $container.find('form');

  $form.on('submit', function(e) {
    $.ajax({
      type: $form.attr('method'),
      url: $form.attr('action'),
      data: $form.serialize(),
      success: function(xhr, status, error) {
        var json = xhr;
        if(json != undefined && json.length > 0){
            for(var i = 0; i < json.length; i++ ){
                var message = json[i].defaultMessage;
                var field = json[i].field;
                if(field == "subscriptionID"){
                    $form.append("<ul class='parsley-errors-list filled padding--top--tiny text-center'><li class='parsley-required'>"+message+"</li></ul>");
                }else{
                    var $field = $("[name='"+field+"']");
                    $field.addClass("error");
                    var type = $field.attr("type");
                    $field.parent().append("<ul class='parsley-errors-list filled'><li class='parsley-required'>"+message+"</li></ul>");
                }
            }
        }else{
            $container.addClass('success');
        }
      },
      error: function(xhr, status, error) {
            var wrongErrorText = "Sorry, er is iets misgegaan. Probeer het s.v.p. opnieuw.";
            $form.append("<ul class='parsley-errors-list filled padding--top--tiny text-center'><li class='parsley-required'>"+wrongErrorText+"</li></ul>");
      }
    });
    return false;
  });
};

};
