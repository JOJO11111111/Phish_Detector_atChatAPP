$(document.body).on('click', '#select-location', function(){
	delete_cookie('city_name');
	delete_cookie('state_name');
	delete_cookie('country_name');
	if($("#city-change").val()){
		var city_id=$("#city-change option:selected").text();
		var selectd=$("#city-change").val();
		
		var selected_data=selectd.split("-");
		var selected_type = selected_data[0];
		var selected_loc_id = selected_data[1];
		var selected_loc_name = selected_data[2];
		
		createCookie('selected_type', selected_type, 90);
		createCookie('selected_loc_id', selected_loc_id, 90);
		createCookie('selected_loc_name', selected_loc_name, 90);

		createCookie('city_id', city_id, 90);	
	}	
	else if($("#state-change").val()){
		createCookie('state_id', $("#state-change").val(), 90); 
		delete_cookie('city_id');
		delete_cookie('city_name');
		delete_cookie('country_id');
		delete_cookie('country_name');
	}else if($("#country-change").val()){
		createCookie('country_id', $("#country-change").val(), 90);
		delete_cookie('city_id');
		delete_cookie('city_name');
		delete_cookie('state_id');
		delete_cookie('state_name');
	}
		

	//location.reload(true);
	document.location.href = document.location + '?r=t';
});

/* CUSTOM FUNCTIONS */
function createCookie(name, value, days) {
    var expires;
    var cookie_path;
	var path = "/";

    if (days) {
        var date = new Date();
        date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
        expires = "; expires=" + date.toUTCString();
    } else {
        expires = "";
    }

	if (path != '') {
		cookie_path = "; path=" + path;
	} else {
		cookie_path = "";
	}

    document.cookie = name + "=" + value + expires + cookie_path;
}
function delete_cookie(name) {
	createCookie(name, "", -100);
}
