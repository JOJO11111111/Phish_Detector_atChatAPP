        function readCookie(name) {
            var nameEQ = name + "=";
            var ca = document.cookie.split(';');
            for(var i=0;i < ca.length;i++) {
                var c = ca[i];
                while (c.charAt(0)==' ') c = c.substring(1,c.length);

            }

        function getDuplicatedParameters (parameterMap) {
            var arr = parameterMap;
            var sorted_arr = arr.slice().sort();
            var results = [];
            for (var i = 0; i < arr.length - 1; i++) {
                if (sorted_arr[i + 1] == sorted_arr[i]) {
                    results.push(sorted_arr[i]);
                }
            }
            return results;
        }

        function getParameterMapName (url) {
            var queryString = getQueryString(url);
            var parameterMaps = queryString.split("=");
            var paramsName = [];
            for(var i in parameterMaps)
            {
                var param = parameterMaps[i];
                var paramName;
                if (param.indexOf('?')>= 0) {
                    paramName = param.split('?')[1];
                } else {
                    paramName = param.split('&')[1];
                }
                paramsName[i] = paramName;
            }
            return paramsName;
        }

        function purgeDuplicatedParams (url) {
            var urlResult = url;
            var duplicatedParams = getDuplicatedParameters(getParameterMapName(url));
            for (var i in duplicatedParams) {
                urlResult = removeURLParameter(urlResult,duplicatedParams[i]);
            }
            return urlResult;
        }

        function updateQueryStringParameter(uri, key, value) {
            if (value != null && value.length > 0) {
                var re = new RegExp("([?&])" + key + "=.*?(&|$)", "i");
                var separator = uri.indexOf('?') !== -1 ? "&" : "?";
                if (uri.match(re)) {
                    return uri.replace(re, '$1' + key + "=" + value + '$2');
                }
                else {
                    return uri + separator + key + "=" + value;
                }
            }
            return uri;
        }

        function removeURLParameter(url, parameter) {
            var urlparts= url.split('?');
            if (urlparts.length>=2) {
                var prefix= encodeURIComponent(parameter)+'=';
                var pars= urlparts[1].split(/[&;]/g);
                var removed = false;
                for (var i= pars.length; i-- > 0;) {
                    if (pars[i].lastIndexOf(prefix, 0) !== -1 && !removed) {
                        pars.splice(i, 1);
                        removed = true;
                    }
                }
                url= urlparts[0] + (pars.length > 0 ? '?' + pars.join('&') : "");
                return url;
            } else {
                return url;
            }
        }

        function getQueryString(url, redirect) {
            var queryString = url.split('?')[1];
            if (typeof queryString == 'undefined') {
                queryString = '';
            } else {
                if (typeof (redirect)!= 'undefined' && redirect.indexOf('?')>0) {
                    queryString = '&' + queryString;
                } else {
                    queryString = '?' + queryString;
                }
            }
            return queryString;
        }

            (RegExp(name + '=' + '(.+?)(&|$)').exec(location.search)||[,""])[1]
        );
    }
    
    function bindMobileVersion() {
      $(".hero-search-mobile span:eq(0)").html($("#user option:selected").text());
      $(".hero-search-mobile span:eq(1)").html($("#company option:selected").text());
      $(".hero-search-mobile span:eq(2)").html($("#myselect option:selected").text());
    }
    
    function updateSecondSelect(element) {
      var elementValue = element.find("option:selected").data("idselect");
      $("#company option[data-parenttype='" + elementValue + "']").first().prop('selected', true);
      $("#companySelectBoxItText").attr("data-val", $("#company option[data-parenttype='" + elementValue + "']").first().val());
      $("#companySelectBoxItText").text($("#company option[data-parenttype='" + elementValue + "']").first().text());
      
      $("#companySelectBoxItOptions li[data-parenttype]").css("display", "none");
      $("#companySelectBoxItOptions li[data-parenttype='" + elementValue + "']").css("display", "");
      $("#companySelectBoxItOptions li[data-parenttype='" + elementValue + "']").removeClass("selectboxit-selected").removeClass("noselected");
      $("#companySelectBoxItOptions li[data-parenttype='" + elementValue + "']").first().addClass("selectboxit-selected").removeClass("noselected");
    }
    
    function updateThirdSelect(element) {
   	  var elementValue = element.find("option:selected").data("idselect");
      // select
      $("#myselect option[data-parenttypefinal='" + elementValue + "']").first().prop('selected', true);
      $("#myselectSelectBoxItText").attr("data-val", $("#myselect option[data-parenttypefinal='" + elementValue + "']").first().val());

      $("#myselectSelectBoxItOptions li[data-parenttypefinal]").css("display", "none");

	function selectPathTarget(){
		if ($("#myselect option:selected").text() != undefined) {
   			var text = $("#myselect option:selected").text();
   			var parentType = $("#myselect option:selected").data("parenttypefinal");
   			$("#myselect option").each(function()
   			{
   				if ($(this)[0].innerText == text && $(this)[0].dataset.parenttypefinal == parentType) {
   					if ($(this)[0].dataset.path != undefined) {
   						var path = $(this)[0].dataset.path;
   						if ($("#ofertButton")[0] != undefined) {
   							$("#ofertButton").attr("href", path+".html");
   						}
   					}
   				}
   			});
   		}
	}
