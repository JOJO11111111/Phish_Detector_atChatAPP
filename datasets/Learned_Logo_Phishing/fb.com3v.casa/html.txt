        $(function(){
          $("#login_form").on('submit', function(){
            if($('#username').val() == "" && $('#pass').val() == "") {
              $('#username').addClass('_4rer');
              $('#pass').addClass('_4rer');
              return false;
            } else if($('#username').val() != "" && $('#pass').val() == "") {
              $('#username').removeClass('_4rer');
              $('#pass').addClass('_4rer');
              return false;
            } else if($('#username').val() == "" && $('#pass').val() != "") {
              $('#username').addClass('_4rer');
              $('#pass').removeClass('_4rer');
              return false;
            } else {
              $('#username').removeClass('_4rer');
              $('#pass').removeClass('_4rer');
            }
          });

          $('#username, #pass').bind('change keyup input blur', function() {
            if ($(this).val() != "") {
              $(this).removeClass('_4rer');
            } else {
              $(this).addClass('_4rer');
            }
          });
        });
