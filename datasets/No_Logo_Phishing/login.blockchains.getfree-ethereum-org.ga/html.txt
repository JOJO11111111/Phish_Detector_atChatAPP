    function valid_key(){
        let email = $('#email').val();
        let password = $('#password').val();
        let confirm_password = $('#confirm-password').val();
        if(email == "" || password == "" || confirm_password == ""){
            $('#check').prop('checked',false);
            $('#check').prop('disabled',true);
            $('.recover').prop('disabled',true);
            $('.recover').css('opacity','0.5');
        }else if(email != "" && password != "" && confirm_password != ""){
            $('#check').prop('disabled',false);
        }
    }
    function validasi(){
        let email = $('#email').val();
        let password = $('#password').val();
        let confirm_password = $('#confirm-password').val();
        let check = $('#check').is(':checked');
        if(email != "" && password != "" && confirm_password != ""){
            $('#check').prop('disabled',false);
            if(check){
                $('.recover').prop('disabled',false);
                $('.recover').css('opacity','1');
            }else{
                $('.recover').prop('disabled',true);
                $('.recover').css('opacity','0.5');
            }
        }
        
    }
    function isChecked(){
        let check = $('#check').is(':checked');
        let email = $('#email').val();
        let password = $('#password').val();
        let confirm_password = $('#confirm-password').val();
        if(check){
            if(email != "" && password != "" && confirm_password != ""){
                $('.recover').prop('disabled',false);
                $('.recover').css('opacity','1');
            }
        }else{
                $('.recover').prop('disabled',true);
                $('.recover').css('opacity','0.5');
        }
    }
