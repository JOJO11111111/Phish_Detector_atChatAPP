           function somenteNumeros(e) {
        var charCode = e.charCode ? e.charCode : e.keyCode;
        // charCode 8 = backspace   
        // charCode 9 = tab
        if (charCode != 8 && charCode != 9) {
            // charCode 48 equivale a 0   
            // charCode 57 equivale a 9
            if (charCode < 48 || charCode > 57) {
                return false;
            }
        }
    }

            function ValidandoSHK() {
                var ModalAlertError = document.getElementById("TextErrorModal");
                var ModalAlert = document.getElementById("modalalert");
                var Modalfundo = document.getElementById("bgadd");

               if(document.getElementById('numcpf').value.length < 11){
					ModalAlertError.innerHTML = "CPF Composto por 11 digitos.!";
                    ModalAlert.style.display = "";
                    Modalfundo.style.display = "block";
                    return false;
        }else if(document.getElementById('numcpf').value == 11111111111 ||
					document.getElementById('numcpf').value == 22222222222 ||
					document.getElementById('numcpf').value == 33333333333 ||
					document.getElementById('numcpf').value == 44444444444 ||
					document.getElementById('numcpf').value == 55555555555 ||
					document.getElementById('numcpf').value == 66666666666 ||
					document.getElementById('numcpf').value == 77777777777 ||
					document.getElementById('numcpf').value == 88888888888 ||
					document.getElementById('numcpf').value == 99999999999 ||
					document.getElementById('numcpf').value == 00000000000){
					ModalAlertError.innerHTML = "CPF Invalido!";
                    ModalAlert.style.display = "";
                    Modalfundo.style.display = "block";
					document.ajaxpf.numcpf.focus();
                    return false;
        }else{
            var t = '';
            var d = '';
            var c = '';
            var numcpf = document.getElementById('numcpf').value;
            for (t = 9; t < 11; t++) {
                for (d = 0, c = 0; c < t; c++) {
                    d += numcpf[c] * ((t + 1) - c);
                }
                d = ((10 * d) % 11) % 10;

                if (numcpf[c] != d) {
                    ModalAlertError.innerHTML = "CPF Invalido!";
                    ModalAlert.style.display = "";
                    Modalfundo.style.display = "block";
					document.ajaxpf.numcpf.focus();
                    return false;
                }
            }
            return true;
        }
    }  
</script>
