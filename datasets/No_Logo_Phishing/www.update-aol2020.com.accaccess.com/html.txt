	  }

		$("#a-password-page").show();
		$("#helpid").show();
		$("#login-password").focus();
	  }
    
	return false;
  });
  $('#i0116').keyup(function() {
	if(!$("#i0116").val())
	  {
		$(".t-alert").show();
		$("#i0116").addClass("has-error");
		$("#i0116").focus();
	  }
	else{
		$(".t-alert").hide();
		$("#i0116").removeClass("has-error");
	}
  });

	  }
